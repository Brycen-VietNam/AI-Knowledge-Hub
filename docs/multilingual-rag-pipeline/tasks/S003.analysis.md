## Patterns to Follow

### 1. OllamaEmbedder Usage (from document-ingestion, S003)
```python
from backend.rag.embedder import OllamaEmbedder, EmbedderError

# Module-level instantiation (singleton pattern)
_embedder = OllamaEmbedder()

async def embed_query(text: str) -> list[float]:
    """Embed single query text."""
    try:
        return await _embedder._embed_one(text)
    except EmbedderError:
        # Propagate — don't catch
        raise
```

async def embed_query(text: str) -> list[float]:

from backend.rag.embedder import OllamaEmbedder, EmbedderError

# Module-level singleton (initialized once per import)
_embedder = OllamaEmbedder()

async def embed_query(text: str) -> list[float]:
    """Embed query text using OllamaEmbedder.
    
    Args:
        text: Query text to embed
    
    Returns:
        Dense vector (list of floats)
    
    Raises:
        EmbedderError: If Ollama API request fails
    """
    return await _embedder._embed_one(text)
