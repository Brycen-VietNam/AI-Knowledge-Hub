import pytest
import uuid
from sqlalchemy.ext.asyncio import AsyncSession

from backend.rag.search import search
from backend.rag.tokenizers.exceptions import LanguageDetectionError
from backend.db.models.document import Document
from backend.db.models.embedding import Embedding

@pytest.fixture(scope="function")
async def seed_multilingual_documents(db_session: AsyncSession):
    """Insert test documents with embeddings across ja, en, vi, ko."""
    docs = {
        "ja": Document(
            id=uuid.uuid4(),
            title="日本語テスト",
            content="これは日本語のテストです。",
            user_group_id=1,
            lang="ja",
        ),
        "en": Document(
            id=uuid.uuid4(),
            title="English Test",
            content="This is an English test document.",
            user_group_id=1,
            lang="en",
        ),
        "vi": Document(
            id=uuid.uuid4(),
            title="Tiếng Việt",
            content="Đây là một tài liệu kiểm tra tiếng Việt.",
            user_group_id=1,
            lang="vi",
        ),
        "public": Document(
            id=uuid.uuid4(),
            title="Public Document",
            content="This document has no user_group_id (public).",
            user_group_id=None,
            lang="en",
        ),
    }
    
    for doc in docs.values():
        db_session.add(doc)
    
    await db_session.commit()
    
    # Insert embeddings for each document
    # (Use existing insert_embeddings() or manual Embedding rows)
    embeddings = [
        Embedding(
            doc_id=docs["ja"].id,
            chunk_index=0,
            text="これは日本語のテストです。",
            lang="ja",
            user_group_id=1,
            embedding=[0.1] * 768,  # Fixed vector for testing
        ),
        # ... repeat for en, vi, public
    ]
    
    for emb in embeddings:
        db_session.add(emb)
    
    await db_session.commit()
    
    yield docs
    
    # Cleanup: delete seeded documents
    for doc in docs.values():
        await db_session.delete(doc)
    await db_session.commit()

@pytest.mark.integration
async def test_search_all_languages_return_results(db_session: AsyncSession, seed_multilingual_documents):
    """Verify search() returns results for each language."""
    docs = seed_multilingual_documents
    
    # Test Japanese detection and retrieval
    results_ja = await search(
        query="日本語",
        user_group_ids=[1],
        session=db_session,
        lang=None,  # Auto-detect
    )
    assert len(results_ja) >= 1, "Should find Japanese documents"
    
    # Test English retrieval
    results_en = await search(
        query="English",
        user_group_ids=[1],
        session=db_session,
        lang="en",
    )
    assert len(results_en) >= 1, "Should find English documents"
    
    # Test Vietnamese retrieval
    results_vi = await search(
        query="Tiếng Việt",
        user_group_ids=[1],
        session=db_session,
        lang="vi",
    )
    assert len(results_vi) >= 1, "Should find Vietnamese documents"

@pytest.mark.integration
async def test_search_rbac_filter(db_session: AsyncSession, seed_multilingual_documents):
    """Verify RBAC: empty user_group_ids returns only public documents."""
    docs = seed_multilingual_documents
    
    results = await search(
        query="document",
        user_group_ids=[],  # No groups → only public
        session=db_session,
        lang="en",
    )
    
    # All results should have user_group_id = None (public)
    for result in results:
        assert result.user_group_id is None, "RBAC filter failed"

@pytest.mark.integration
async def test_search_language_detection_error_on_gibberish(db_session: AsyncSession):
    """Verify LanguageDetectionError for empty/gibberish query."""
    with pytest.raises(LanguageDetectionError):
        await search(
            query="",  # Empty query
            user_group_ids=[1],
            session=db_session,
            lang=None,  # Auto-detect will fail
        )

@pytest.mark.integration
async def test_search_lang_override_ja(db_session: AsyncSession, seed_multilingual_documents):
    """Verify explicit lang='ja' override skips auto-detection."""
    docs = seed_multilingual_documents
    
    # Search with English query but lang="ja"
    results = await search(
        query="English",
        user_group_ids=[1],
        session=db_session,
        lang="ja",  # Override: treat as Japanese
    )
    
    # Should not error, but results may be empty or matched by Japanese tokenizer
    assert isinstance(results, list), "Should return list[RetrievedDocument]"
