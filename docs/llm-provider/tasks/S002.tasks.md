```python
# backend/rag/llm/ollama.py
import os
from pathlib import Path
import httpx
from .base import LLMProvider, LLMResponse
from .exceptions import LLMError, NoRelevantChunksError

_PROMPT_TEMPLATE = (Path(__file__).parent / "prompts" / "answer.txt").read_text()

class OllamaAdapter(LLMProvider):
    def __init__(self):
        self._base_url = os.getenv("OLLAMA_BASE_URL", "http://localhost:11434")
        self._model = os.getenv("LLM_MODEL", "llama3")

    async def complete(self, prompt: str, context_chunks: list[str]) -> LLMResponse:
        if not context_chunks:
            raise NoRelevantChunksError("No relevant chunks — cannot generate answer (C014)")
        filled = _PROMPT_TEMPLATE.format(
            context="\n".join(context_chunks), question=prompt
        )
        try:
            async with httpx.AsyncClient() as client:
                resp = await client.post(
                    f"{self._base_url}/api/generate",
                    json={"model": self._model, "prompt": filled, "stream": False},
                    timeout=5.0,
                )
                resp.raise_for_status()
                data = resp.json()
        except Exception as exc:
            raise LLMError(str(exc)) from exc
        return LLMResponse(
            answer=data["response"],
            sources=context_chunks,
            confidence=0.9,  # sentinel — Ollama has no logprobs (D06)
            provider="ollama",
            model=self._model,
            low_confidence=0.9 < 0.4,  # always False for sentinel
        )
```

**review_criteria:**
- `async httpx.AsyncClient` — NOT sync (D07 — critical)
- `NoRelevantChunksError` raised before any HTTP call when `context_chunks == []`
- All env reads via `os.getenv()` — zero hardcoded values (S005)
- Template loaded at module import time (`_PROMPT_TEMPLATE` module-level) — not per-request
- All exceptions wrapped → `LLMError` — raw httpx never propagated

```python
# backend/rag/llm/openai.py
import os
from pathlib import Path
from .base import LLMProvider, LLMResponse
from .exceptions import LLMError, NoRelevantChunksError

_PROMPT_TEMPLATE = (Path(__file__).parent / "prompts" / "answer.txt").read_text()

class OpenAIAdapter(LLMProvider):
    def __init__(self):
        self._api_key = os.getenv("OPENAI_API_KEY")
        self._model = os.getenv("LLM_MODEL", "gpt-4o-mini")

    async def complete(self, prompt: str, context_chunks: list[str]) -> LLMResponse:
        if not context_chunks:
            raise NoRelevantChunksError("No relevant chunks (C014)")
        import openai  # lazy import (P007)
        filled = _PROMPT_TEMPLATE.format(
            context="\n".join(context_chunks), question=prompt
        )
        try:
            client = openai.OpenAI(api_key=self._api_key)
            resp = client.chat.completions.create(
                model=self._model,
                messages=[{"role": "user", "content": filled}],
            )
        except Exception as exc:
            raise LLMError(str(exc)) from exc
        content = resp.choices[0].message.content
        logprobs = resp.choices[0].logprobs
        confidence = float(logprobs.content[0].logprob) if logprobs else 0.9
        return LLMResponse(
            answer=content,
            sources=context_chunks,
            confidence=confidence,
            provider="openai",
            model=self._model,
            low_confidence=confidence < 0.4,
        )
```

**review_criteria:**
- `OPENAI_API_KEY` via `os.getenv()` — never hardcoded (S005)
- `import openai` is lazy inside `complete()` — not at module level (P007)
- `NoRelevantChunksError` before any SDK call
- Logprobs → confidence; fallback sentinel 0.9 when logprobs unavailable (D06)
- `low_confidence = confidence < 0.4` (C014)

```python
# backend/rag/llm/claude.py
import os
from pathlib import Path
from .base import LLMProvider, LLMResponse
from .exceptions import LLMError, NoRelevantChunksError

_PROMPT_TEMPLATE = (Path(__file__).parent / "prompts" / "answer.txt").read_text()

class ClaudeAdapter(LLMProvider):
    def __init__(self):
        self._api_key = os.getenv("ANTHROPIC_API_KEY")
        self._model = os.getenv("LLM_MODEL", "claude-haiku-4-5-20251001")

    async def complete(self, prompt: str, context_chunks: list[str]) -> LLMResponse:
        if not context_chunks:
            raise NoRelevantChunksError("No relevant chunks (C014)")
        import anthropic  # lazy import (P007)
        filled = _PROMPT_TEMPLATE.format(
            context="\n".join(context_chunks), question=prompt
        )
        # Route A prompt caching: answer.txt is stable prefix; {question}+{context} are volatile suffix
        try:
            client = anthropic.Anthropic(api_key=self._api_key)
            msg = client.messages.create(
                model=self._model,
                max_tokens=1024,
                messages=[{"role": "user", "content": filled}],
            )
        except Exception as exc:
            raise LLMError(str(exc)) from exc
        return LLMResponse(
            answer=msg.content[0].text,
            sources=context_chunks,
            confidence=0.9,  # sentinel — Claude API has no logprobs (D06)
            provider="claude",
            model=self._model,
            low_confidence=0.9 < 0.4,  # always False for sentinel
        )
```

- `ANTHROPIC_API_KEY` via `os.getenv()` — never hardcoded (S005)

- `import anthropic` is lazy inside `complete()` — not at module level (P007)
