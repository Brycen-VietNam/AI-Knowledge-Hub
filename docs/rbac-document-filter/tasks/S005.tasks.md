```python
# backend/api/routes/query.py
from uuid import uuid4
from fastapi import APIRouter, BackgroundTasks, Depends
from pydantic import BaseModel, Field
from backend.auth.dependencies import verify_token
from backend.auth.types import AuthenticatedUser

class QueryRequest(BaseModel):
    query: str = Field(..., max_length=512)
    top_k: int = Field(default=10, ge=1, le=100)

class QueryResult(BaseModel):
    doc_id: str
    chunk_index: int
    score: float
    is_public: bool
    content: str | None = None

class QueryResponse(BaseModel):
    request_id: str
    results: list[QueryResult]

router = APIRouter()

@router.post("/v1/query", response_model=QueryResponse,
             dependencies=[Depends(verify_token)])
async def query_documents(
    body: QueryRequest,
    background_tasks: BackgroundTasks,
    user: AuthenticatedUser = Depends(verify_token),
) -> QueryResponse:
    request_id = str(uuid4())
    return QueryResponse(request_id=request_id, results=[])
```

### T003: RBAC group resolution + retrieve() call + audit log

**Status**: TODO

**Change description**
Replace the stub `query_documents` body with full RBAC implementation:

1. **Group resolution** — already done by `verify_token`: `user.user_group_ids: list[int]` is populated
   by `verify_api_key` (direct from `api_keys.user_group_ids`) and `verify_oidc_token` (group name→ID lookup).
   No extra resolution needed in the route — pass `user.user_group_ids` directly.

2. **Retrieval** — call `retrieve()` from `backend.rag.retriever`:
   ```python
   docs = await retrieve(
       query_embedding=await embed(body.query),
       user_group_ids=user.user_group_ids,
       top_k=body.top_k,
       session=db,
   )
   ```
   Wrap in `asyncio.wait_for(..., timeout=1.8)` — raise `QueryTimeoutError` → 504 response.

3. **Audit log** — non-blocking background task:
   ```python
   query_hash = hashlib.sha256(body.query.encode()).hexdigest()
   background_tasks.add_task(_write_audit, db, user.user_id, docs, query_hash)
   ```

4. **Response** — map `RetrievedDocument` → `QueryResult`:
   ```python
   results = [
       QueryResult(
           doc_id=str(d.doc_id),
           chunk_index=d.chunk_index,
           score=d.score,
           is_public=d.user_group_id is None,
           content=d.content,
       )
       for d in docs
   ]
   return QueryResponse(request_id=request_id, results=results)
   ```

5. **0-group users**: `user_group_ids=[]` → `retrieve()` returns only public docs → `200 {"results": [public_docs]}`. Not 403.

6. **_write_audit helper** (module-private):
   ```python
   async def _write_audit(session, user_id, docs, query_hash):
       from backend.db.models.audit_log import AuditLog
       async with session.begin():
           for doc in docs:
               session.add(AuditLog(user_id=user_id, doc_id=doc.doc_id, query_hash=query_hash))
   ```

Add `db: AsyncSession = Depends(get_db)` parameter to route.
