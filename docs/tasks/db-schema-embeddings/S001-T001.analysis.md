## Recommended Implementation for T001

```sql
-- 001_create_core_schema.sql
-- Requires: PostgreSQL >= 13
-- Creates: user_groups, documents, embeddings, audit_logs

CREATE TABLE IF NOT EXISTS user_groups (
    id          INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    name        TEXT        NOT NULL,
    created_at  TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE IF NOT EXISTS documents (
    id              UUID        PRIMARY KEY DEFAULT gen_random_uuid(),
    title           TEXT        NOT NULL,
    lang            CHAR(2)     NOT NULL,
    user_group_id   INT         NOT NULL REFERENCES user_groups(id),
    created_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT now()
    -- Note: updated_at must be set by application layer on UPDATE
);

CREATE TABLE IF NOT EXISTS embeddings (
    id              UUID        PRIMARY KEY DEFAULT gen_random_uuid(),
    doc_id          UUID        NOT NULL REFERENCES documents(id) ON DELETE CASCADE,
    chunk_index     INT         NOT NULL,
    lang            CHAR(2)     NOT NULL,
    user_group_id   INT         NOT NULL,   -- denormalized for RBAC WHERE clause (R001)
    created_at      TIMESTAMPTZ NOT NULL DEFAULT now()
    -- Note: embedding vector(1024) column added in migration 002 after pgvector extension
);

CREATE TABLE IF NOT EXISTS audit_logs (
    id          UUID        PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id     TEXT        NOT NULL,   -- placeholder; auth schema TBD
    doc_id      UUID        NOT NULL REFERENCES documents(id),
    query_hash  TEXT        NOT NULL,
    accessed_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- ROLLBACK
-- DROP TABLE IF EXISTS audit_logs CASCADE;
-- DROP TABLE IF EXISTS embeddings CASCADE;
-- DROP TABLE IF EXISTS documents CASCADE;
-- DROP TABLE IF EXISTS user_groups CASCADE;
```

**Key call-outs**:
- `embeddings.doc_id` uses `ON DELETE CASCADE` — delete document → orphaned chunks auto-removed
- `embeddings.user_group_id` has no FK constraint (denormalized INT, avoids JOIN overhead at query time)
