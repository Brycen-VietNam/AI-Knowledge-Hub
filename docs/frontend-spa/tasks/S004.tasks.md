### T001: QueryHistoryItem type + history state in queryStore

**Status**: DONE

**File(s)**
- `frontend/src/store/queryStore.ts` — modify: add interface + history state + 3 actions
- `frontend/tests/store/queryStore.test.ts` — modify: add tests for history actions

**Change description**
Add `QueryHistoryItem` interface (id, query, answer, citations, timestamp) and three new
actions to the existing Zustand store: `addHistory` (max 20, newest first), `clearHistory`,
and `selectHistory` (restores query + result into current state). `reset()` does NOT clear
history (session-only; clears on logout only).

**Key changes**
```ts
// Interface
export interface QueryHistoryItem {
  id: string           // crypto.randomUUID()
  query: string
  answer: string
  citations: Citation[]
  timestamp: Date
}

// State additions
history: QueryHistoryItem[]
addHistory: (query: string, answer: string, citations: Citation[]) => void
clearHistory: () => void
selectHistory: (item: QueryHistoryItem) => void

// addHistory implementation
addHistory: (query, answer, citations) => set((s) => ({
  history: [{ id: crypto.randomUUID(), query, answer, citations, timestamp: new Date() },
            ...s.history].slice(0, 20),
})),

// clearHistory
clearHistory: () => set({ history: [] }),

// selectHistory — restore into current query + result state
selectHistory: (item) => set({
  query: item.query,
  result: { answer: item.answer, citations: item.citations, confidence: 0 },
  error: null,
}),
```

**Key changes**
```tsx
// HistoryItem.tsx
interface Props {
  item: QueryHistoryItem
  onSelect: (item: QueryHistoryItem) => void
}

// CJK-safe truncation (NOT str.slice — breaks multi-byte chars)
function truncate(str: string, len = 60): string {
  const chars = [...str]
  return chars.length > len ? chars.slice(0, len).join('') + '…' : str
}

// Timestamp: HH:mm via toLocaleTimeString('ja-JP', { hour:'2-digit', minute:'2-digit' })
// or format manually: hours.padStart(2,'0') + ':' + mins.padStart(2,'0')

// Render:
<li role="listitem" onClick={() => onSelect(item)}>
  <span data-testid="history-query">{truncate(item.query)}</span>
  <time data-testid="history-time">{HH:mm}</time>
</li>
```

**i18n**: none required (timestamp is locale-neutral HH:mm; query is user text)

**Status**: DONE

- `frontend/tests/components/history/HistoryPanel.test.tsx` — create

**Change description**
Container panel that renders the list of `HistoryItem` entries and a "Clear" button.
The panel returns `null` (renders nothing) when `history.length === 0`. Uses
`useQueryStore` for data and actions.

**Key changes**
```tsx
// HistoryPanel.tsx
import { useQueryStore } from '../../store/queryStore'
import { HistoryItem } from './HistoryItem'
import { useTranslation } from 'react-i18next'

export function HistoryPanel() {
  const { history, clearHistory, selectHistory } = useQueryStore()
  const { t } = useTranslation()

  if (history.length === 0) return null   // AC Q10 — hidden when empty

  return (
    <aside aria-label={t('history.title')}>
      <h2>{t('history.title')}</h2>
      <button onClick={clearHistory}>{t('history.clear')}</button>
      <ul>
        {history.map((item) => (
          <HistoryItem key={item.id} item={item} onSelect={selectHistory} />
        ))}
      </ul>
    </aside>
  )
}
```
